//TargetLevel 涵瘸]ヘ夹单
//NowLevel Oヘe旱ク拧A讽ヘe旱ク=ヘ夹单糯N|_u

[enable]
registersymbol(TargetLevel)
registersymbol(NowLevel)
alloc(CheckLevel,64)
label(TargetLevel)
label(NowLevel)
label(CheckLevelRet)
label(Return)

TargetLevel:
  dd  00
NowLevel:
  dd  00

CheckLevel:
    mov [NowLevel], eax
    cmp [TargetLevel], 00
    jz Return
    cmp eax, [TargetLevel]
    jne Return
    push 0
    call ExitProcess
    jmp Return
Return:
    cmp eax,000000C8
    jmp CheckLevelRet

00647357:
jmp CheckLevel
CheckLevelRet:
[disable]
00647357:
cmp eax,000000C8

unregistersymbol(TargetLevel)
unregistersymbol(NowLevel)
dealloc(CheckLevel)
